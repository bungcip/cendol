// contains all builtins and other popular c types

// Basic types
void => FLAG_CANONICAL
bool => FLAG_CANONICAL | FLAG_INTEGER | FLAG_UNSIGNED
char => FLAG_CANONICAL | FLAG_INTEGER | FLAG_SIGNED
short => FLAG_CANONICAL| FLAG_INTEGER | FLAG_SIGNED
int => FLAG_CANONICAL | FLAG_INTEGER | FLAG_SIGNED
long => FLAG_CANONICAL | FLAG_INTEGER | FLAG_SIGNED
float => FLAG_CANONICAL | FLAG_FLOATING
double => FLAG_CANONICAL | FLAG_FLOATING
